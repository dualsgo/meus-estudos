# A condição é avaliada como True ou False. 
if condição == True:
# Caso a condição seja True, o bloco de código dentro do if é executado.
    
    # Bloco True: Observem a indentação, que é a forma como o Python identifica o início e o fim de um bloco de código.

# Caso a condição seja False, o bloco de código dentro do else é executado.

    print("Bloco de código que será executado caso a condição não seja atendida.")

# Instrução fora do bloco
print('Fim do programa.')

# Observação: Altere a variável hora para testar as condições

hora = 10 # Define a hora atual

# Verifica se a hora está entre 0 e 23, representando as 24 horas do dia

# if 0 <= hora and hora <= 23: -> Aqui, a condição é verificada de forma explícita as duas possibilidades. A condição é atendida se a hora for maior ou igual a 0 e menor ou igual a 23, mas podemos simplificar a condição para if 0 <= hora <= 23: que é mais legível e mais comum em Python.
if 0 <= hora <= 23:
    # Note que entramos em um novo bloco de código, pois a condição foi atendida. Todo código dentro do bloco será executado. Se forem novas estruturas condicionais, elas serão avaliadas e também terão seus blocos de código mais internos.
        
    # Se a hora for menor que 12, mas maior ou igual a 5 exibe 'Bom dia!'. Em outras palavras, se a hora estiver entre 5 e 11, exibe 'Bom dia!'.
    if 5 <= hora <= 11:  # Manhã entre 5 e 11
        print('Bom dia!')
        
        # Dentro do bloco de código do if, podemos ter outras estruturas condicionais. Isso é chamado de estrutura condicional aninhada. Recaptulando, se a execução do código chegou até aqui, significa que a hora está entre 0 e 23, e que está ente 5 e 11. Então, podemos fazer novas verificações dentro desse intervalo.
        
        # Se a hora for igual a 8, exibe 'Hora do café da manhã!'
        if hora == 8:
            print('Hora do café da manhã!')
            
        # Se a hora for igual a 10, exibe 'Hora do desenho!'
        elif hora == 10:
            print('Hora do desenho!')
            
        # Se estiver entre 5 e 11, mas não for 8 nem 10, não exibe nada, pois não definimos uma ação para esses casos. 
    
    # Se a hora for maior que 12, mas menor ou igual a 17, exibe 'Boa tarde!'
    elif 12 <= hora <= 17:  # Tarde entre 12 e 17
        print('Boa tarde!')
        
    # Se a hora for maior que 17, mas menor ou igual 23, exibe 'Boa noite!'
    elif 18 <= hora <= 23:  # Noite entre 18 e 23
        print('Boa noite!')

    # Se a hora for maior que 23, mas menor que 5, exibe 'Boa madrugada!'
    else:  # Madrugada será entre 0 e 4
        print('Boa madrugada!')
        
    # Basicamente, sem a estrutura aninhada, se nenhuma das condições acima fosse atendida, exibiria 'Boa madrugada!' para qualquer valor inteiro digitado.
        
# Se a hora não estiver entre 0 e 23, exibe 'Hora inválida!'        
else:
    print('Hora inválida!')
Bom dia! Hora do desenho!

# Aqui temos duas estruturas condicionais. A primeira é simples e a segunda é composta
x = 4

# Primeira estrutura condicional
if x < 10:  # condição
    print(f"{x} é menor que 10")   # Executado se a condição for True.

# Segunda estrutura condicional
if x > 10:  # condição
    print(f"{x} é maior que 10")   # Executado se a condição for True.

else: # Se a segunda condição for False. Isso pode causar um erro de lógica.
    print(f"{x} é igual a 10")

# O código acima está sintaticamente correto, mas logicamente pode estar errado. De fato se x vale 4, ele não é maior que 10 e portanto a segunda condição não é atendida. O comportamento do código é executar a instrução dentro do bloco else. O ramo else não possui uma condição para avaliar. Ele é executado caso a condição do if seja False. Nesse caso, a condição do if é False e ele é executao mostrando uma mensagem incorreta
