def contador(i, f, p):
    """
    Faz uma contagem e mostra na tela.
    :param i: INÍCIO da contagem
    :param f: FIM da contagem
    :param p: PASSO daa contagem
    :return: Sem retorno
    """
    c = i
    while c <= f:
        print(f'{c}', end='..')
        c += p
    print('FIM!')


contador(2, 10, 2)
